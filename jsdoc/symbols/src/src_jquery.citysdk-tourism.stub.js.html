124 /**
125  * @name TourismClient
126  * @class jQuery client stub for the CitySDK Tourism WP5
127  * @param data mapping of available capacities in the server
128  */
129 TourismClient = function(data) {
130 	var mapping = mapConvert(data), version;
131 	
132 	/**
133 	 * @private
134 	 */
135 	function mapConvert(data) {
136 		var links = data["citysdk-tourism"], map = [];
137 		for(key in links) {
138 			map[links[key].version] = links[key]["_links"];
139 		}
140 			
141 		return map;
142 	};
143 	
144 	/**
145      * Use a given version
146      * @memberOf TourismClient
147      * @param v version to use
148      */
149 	this.useVersion = function(v) {
150 		version = v;
151 	};
152 	
153 	/**
154      * Gets the available resources for the visited server.
155      * @memberOf TourismClient
156      * @returns {String[]} a mapping of available resources.
157      * @throws {VersionNotAvailableException} thrown if the version was not set or is not available.
158      */
159 	this.getResources = function() {
160 		verifyVersion();
161 		return Object.keys(mapping[version]);
162 	};
163 	
164 	/**
165      * Checks whether the given resource is available or not.
166      * @memberOf TourismClient
167      * @param resource the resource to check for availability.
168      * @returns {boolean} true if available, false otherwise.
169      * @throws {VersionNotAvailableException} thrown if the version was not set or is not available.
170      */
171 	this.hasResource = function(resource) {
172 		verifyVersion();
173 		return resource in mapping[version];
174 	};
175 	
176 	/**
177      * Checks whether the given parameter is available in a resource or not.
178      * @memberOf TourismClient
179      * @param resource the resource to check the parameter.
180      * @param parameter the parameter to check for availability
181      * @returns {boolean} true if available, false otherwise.
182      * @throws {VersionNotAvailableException} thrown if the version was not set or is not available.
183      */
184 	this.hasResourceParameter = function(resource, parameter) {
185 		verifyVersion();
186 		UriTemplate.fromTemplate(mapping[version][resource].href);
187 		return UriTemplate.hasParameter(parameter);
188 	};

358 	 * @private
359 	 */ 
360 	getList = function(parameters, handleData, handleError, resource) {
361 		verifyVersion();
362 		validateResource(resource);
363 		validateParameters(resource, parameters);
364 		makeQueryCall(mapping[version][resource].href, parameters, handleData, handleError);
365 	};

367 	/**
368 	 * @private
369 	 */
370 	getCategorization = function(parameters, handleData, handleError, resource) {
371 		verifyVersion();
372 		validateResource(resource);
373 		validateParameters(resource, parameters);
374 		validateListTerm(parameters['list']);
375 		makeQueryCall(mapping[version][resource].href, parameters, handleData, handleError);
376 	};
377

379 	 * @private
380 	 */ 
381 	getRelation = function(base, id, relation, handleData, handleError, resource) {
382 		verifyVersion();
383 		validateResource(resource);
384 		validateRelationTerm(relation);
385 		var parameters = {
386 			"base": base,
387 			"id": id,
388 			"relation": term
389 		};
390 		makeQueryCall(mapping[version][resource].href, parameters, handleData, handleError);
391 	};

393 	/**
394 	 * @private
395 	 */
396 	getCode = function(code, handleData, handleError, resource) {
397 		verifyVersion();
398 		validateResource(resource);
399 		var parameters = {
400 			"code": code
401 		};
402 		makeQueryCall(mapping[version][resource].href, parameters, handleData, handleError);
403 	};

405 	/**
406 	 * @private
407 	 */
408 	makeQueryCall = function(resource, parameters, handleData, handleError) {
409 		UriTemplate.fromTemplate(resource);
410 		$.support.cors = true;
411 		$.ajax({
412 			headers: { 
413         		Accept : "application/json"
414         	},
415         	crossDomain: true,
416         	dataType: "json",
417 			url: encodeURI(UriTemplate.build(parameters)),
418 			type: 'GET',
419 			processData: true,
420 			success: function(data, textStatus, jqXHR) {
421 				handleData(data, textStatus, jqXHR);
422 			},
423 			error: function(jqXHR, textStatus, errorThrown) {
424 				handleError(jqXHR, textStatus, errorThrown);
425 			}
426 		});
427 	};
428 	
429 	/**
430 	 * @private
431 	 */
432 	makeSingleCall = function(url, handleData, handleError) {
433 		$.support.cors = true;
434 		$.ajax({
435 			headers: { 
436         		Accept : "application/json"
437         	},
438         	crossDomain: true,
439 			url: encodeURI(url),
440 			dataType: "json",
441 			type: 'GET',
442 			success: function(data, textStatus, jqXHR) {
443 				handleData(data, textStatus, jqXHR);
444 			},
445 			error: function(jqXHR, textStatus, errorThrown) {
446 				handleError(jqXHR, textStatus, errorThrown);
447 			}
448 		});
449 	};

451 	/**
452 	 * @private
453 	 */
454 	verifyVersion = function() {
455 		if(version == undefined
456 			|| mapping[version] == undefined) {
457 			throw {
458 				name: "VersionNotAvailableException",
459 		   		level: "Show Stopper",
460 		    	message: version + " is not available"
461 			};
462 		}
463 	};
464 	
465 	/**
466 	 * @private
467 	 */
468 	validateResource = function(resource) {
469 		if(!(resource in mapping[version])) {
470 			throw {
471 				name: "ResourceNotAvailableException",
472 		   		level: "Show Stopper",
473 		    	message: resource + " is not available"
474 			};
475 		}
476 	};
477  	
478 	/**
479 	 * @private
480 	 */
481 	validateParameters = function(resource, parameters) {
482 		UriTemplate.fromTemplate(mapping[version][resource].href);
483 		for(key in parameters) {
484 			if(!UriTemplate.hasParameter(key)) {
485 				throw {
486 		    		name: "InvalidParameterException",
487 		   			level: "Show Stopper",
488 		    		message: key + " parameter is invalid for resource " + resource
489 		   		};
490    			}
491 		}
492 	};
493 	
494 	/**
495 	 * @private
496 	 */
497 	validateListTerm = function(term) {
498 		if(term == undefined 
499 			|| (term != parameterTerms.POIS
500 				&& term != parameterTerms.EVENTS
501 				&& term != parameterTerms.ROUTES)) {
502 			throw {
503 		    	name: "InvalidTermException",
504 		   		level: "Show Stopper",
505 		    	message: term + " is invalid. It should be either poi, event or route"
506 		   	};
507 		}	
508 	};
509 	
510 	/**
511 	 * @private
512 	 */
513 	validateRelationTerm = function(term) {
514 		if(term == undefined) {
515 			throw {
516 		    	name: "InvalidTermException",
517 		   		level: "Show Stopper",
518 		    	message: term + " is invalid. It should be either child or parent."
519 		   	};
520 		}
521 		
522 		if(term == undefined 
523 			|| (term != "child"
524 				&& term != "parent")) {
525 			throw {
526 		    	name: "InvalidTermException",
527 		   		level: "Show Stopper",
528 		    	message: term + " is invalid. It should be either child or parent."
529 		   	};
530 		}
531 	};
532 };

535  * @name TourismVisitor
536  * @class Used to generate a TourismClient stub, after visiting a given URI through @see visitHome.

540 /**
541  * Visits the given URI and returns a Client stub with the available capacities.
542  * @static
543  * @memberOf TourismVisitor
544  * @param uri the wanted URI.
545  * @param handleData the callback containing the response from this function. This callback should receive a TourismClient stub as a parameter.
546  * @param handleError the callback to handle network errors. Its parameters should be a jQuery XMLHttpRequest (jqXHR) object, textStatus and errorThrown. 
547  */
548 TourismVisitor.visit = function(uri, handleData, handleError) {
549 	$.support.cors = true;
550 	$.ajax({
551 		headers: { 
552         	Accept : "application/json"
553         },
554         crossDomain: true,
555         dataType: "json",
556 		url: uri,
557 		type: 'GET',
558 		success: function(data, textStatus, jqXHR) {
559 			handleData(new TourismClient(data), textStatus, jqXHR);
560 		},
561 		error: function(jqXHR, textStatus, errorThrown) {
562 			handleError(jqXHR, textStatus, errorThrown);
563 		}
564 	});
565 };
