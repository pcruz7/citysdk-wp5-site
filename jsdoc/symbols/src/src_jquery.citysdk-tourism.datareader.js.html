144 /**
145  * @private
146  */
147 DataReader.langsAreEqual = function(lang1, lang2) {
148 	if(lang1 == undefined || lang2 == undefined)
149 		return false;
150 	
151 	var parsedLang1 = lang1.replace("-", "_").split("_"),
152 		parsedLang2 = lang2.replace("-", "_").split("_");
153 
154 	return parsedLang1[0] == parsedLang2[0];
155 };

185 /**
186  * Gets the local language and country.
187  * @memberOf DataReader
188  * @returns {String} the local language in the following format lang_COUNTRY (e.g.: pt_PT) or 
189  * just "en" if it was not possible to get it. 
190  */
191 DataReader.getLocalLanguage = function() {
192 	if (navigator.userLanguage)   // Explorer
193 		return navigator.userLanguage.replace("-", "_");
194 	else if (navigator.language)  // FF, Chrome
195 		return navigator.language.replace("-", "_");
196 	else
197 	 	return "en";
198 };

402 /**
403  * Gets the available languages for the POI object.
404  * @static
405  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
406  * @param field specifies the field to be checked (either label or description).
407  * @returns {String[]} an array containing the available languages of a given field.
408  */
409 DataReader.getAvailableLanguages = function(poi, field) {
410 	var values, 
411 		languages = [],
412 		lang, primary;
413 	if(poi == null || poi == undefined)
414 		return languages;
415 		
416 	if(field != 'label' && field != 'description')
417 		return languages;
418 	
419 	values = poi[field];
420 	for(key in values) {
421 		lang = values[key].lang;
422 		if(lang == undefined)
423 			lang = poi.lang;
424 			
425 		primary = lang.replace("-", "_").split("_")[0];
426 		if(!(primary in languages))
427 			languages[primary] = lang;
428 	}
429 	
430 	return languages;
431 };

445 /**
446  * Gets a label description from the POI object with a given term in a given language.
447  * @static
448  * @param poi a single POI object (such as a Point of Interest, Route or Event).
449  * @param term the wanted term (see {@link term}).
450  * @param lang the wanted language.
451  * @returns {String} a label description corresponding to the term in a given language, or en_GB if
452  * the wanted language does not exist or false if both do not exist. 
453  */
454 DataReader.getLabel = function(poi, term, lang) {
455 	if(poi == null || poi == undefined)
456 		return false;
457 	
458 	if(this.defaultLanguage == undefined)
459 		this.setDefaultLanguage("en_GB");
460 	
461 	var labelLang,
462 		poiLang = poi.lang,
463 		labels = poi.label,
464 		defaultValue = null;
465 	
466 	for(key in labels) {
467 		if(labels[key].lang == undefined) 
468 			labelLang = poiLang;
469 		else
470 			labelLang = labels[key].lang;
471 		
472 		if(labels[key].term == term 
473 			&& this.langsAreEqual(labelLang, this.defaultLanguage)) {
474 			defaultValue = labels[key].value;
475 		} else if(labels[key].term == term
476 					&& this.langsAreEqual(labelLang, lang)) {
477 			return labels[key].value;
478 		}
479 	}
480 	
481 	return defaultValue;
482 };

484 /**
485  * Gets a given description from the POI object with a given language.
486  * @memberOf DataReader
487  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
488  * @param lang the wanted language.
489  * @returns {String} a description in a given language, or in en_GB if
490  * the wanted language does not exist or false if both do not exist.
491  */
492 DataReader.getDescription = function(poi, lang) {
493 	if(poi == null || poi == undefined)
494 		return false;
495 
496 	if(this.defaultLanguage == undefined)
497 		this.setDefaultLanguage("en_GB");
498 	
499 	var descriptionLang, poiLang = poi.lang,
500 		descriptions = poi.description,
501 		defaultValue = null;
502 	for(key in descriptions) {
503 		if(descriptions[key].lang == undefined) 
504 			descriptionLang = poiLang;
505 		else
506 			descriptionLang = descriptions[key].lang;
507 			
508 		if((descriptions[key].type == undefined || descriptions[key].type == null)
509 			&& this.langsAreEqual(descriptionLang, this.defaultLanguage)) {
510 			defaultValue = descriptions[key].value;
511 		} else if((descriptions[key].type == undefined || descriptions[key].type == null) 
512 					&& this.langsAreEqual(descriptionLang, lang)) {
513 			return descriptions[key].value;
514 		}
515 	}
516 	
517 	return defaultValue;
518 };

552 /**
553  * @private
554  */
555 DataReader.getValueWithTag = function(poi, lang, tag) {
556 	if(poi == null || poi == undefined)
557 		return false;
558 		
559 	if(this.defaultLanguage == undefined)
560 		this.setDefaultLanguage("en_GB");
561 	
562 	var descriptionLang, 
563 		poiLang = poi.lang,
564 		descriptions = poi.description,
565 		defaultValue = null;
566 	for(key in descriptions) {
567 		if(descriptions[key].lang == undefined) 
568 			descriptionLang = poiLang;
569 		else
570 			descriptionLang = descriptions[key].lang;
571 			
572 		if(descriptions[key].type != undefined
573 			&& descriptions[key].type == tag) {
574 			if(lang == null || this.langsAreEqual(descriptionLang, this.defaultLanguage))
575 				defaultValue = descriptions[key].value;
576 		} else if(descriptions[key].type != undefined
577 					&& descriptions[key].type == tag) {
578 			if(lang == null || this.langsAreEqual(descriptionLang, this.defaultLanguage))
579 				return descriptions[key].value;
580 		}
581 	}
582 	
583 	return defaultValue;
584 };

586 /**
587  * Gets all the thumbnail URI or base-64 bytecode from the POI object.
588  * @memberOf DataReader
589  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
590  * @returns {ImageContent[]} all the thumbnail URI or base-64 bytecode or an empty list.
591  */
592 DataReader.getThumbnails = function(poi) {
593 	var thumbnails = [],
594 		links,
595 		content;
596 	if(poi == null || poi == undefined)
597 		return thumbnails;
598 	
599 	links = poi.link;
600 	for(key in links) {
601 		if(links[key].term == term.LINK_TERM_ICON) {
602 			if(links[key].href != undefined) {
603 				content = new ImageContent(links[key].href);
604 				content.isImgUri(true);
605 				thumbnails.push(content);
606 			} else if (links[key].value != undefined) {
607 				content = new ImageContent(links[key].value);
608 				content.isImgByteCode(true);
609 				thumbnails.push(content);
610 			}
611 		}
612 	}
613 	
614 	return thumbnails;
615 };

617 /**
618  * Gets the point from the location of the POI object with a given term.
619  * @memberOf DataReader
620  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
621  * @param term the wanted term.
622  * @returns {PointContent[]} an array containing all points with the given term or an empty array.
623  */
624 DataReader.getLocationPoint = function(poi, term) {
625 	var list = [], 
626 		location, 
627 		point, 
628 		data, 
629 		points;
630 	if(poi == null || poi == undefined)
631 		return list;
632 		
633 	location = poi.location;
634 	if(location.point != undefined) {
635 		points = location.point;
636 		for(key in points) {
637 			if(points[key].term == term) {
638 				data = []; 
639 				data = points[key].Point.posList.split(" ");
640 				point = new PointContent(data[0], data[1]);
641 				list.push(point);
642 			}
643 		}
644 	}
645 	
646 	return list;
647 };

652  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
653  * @param term the wanted term.
654  * @returns {LineContent[]} an array containing all lines with the given term or an empty array.
655  */
656 DataReader.getLocationLine = function(poi, term) {
657 	var list = [],
658 		location,
659 		line,
660 		data, 
661 		point1,
662 		point2, 
663 		lines;
664 	if(poi == null || poi == undefined)
665 		return list;
666 		
667 	location = poi.location;
668 	if(location.line != undefined) {
669 		lines = location.line;
670 		for(key in lines) {
671 			if(lines[key].term == term) {
672 				data = [];
673 				point1 = [];
674 				point2 = [];
675 				data = lines[key].LineString.posList.split(",");
676 				point1 = data[0].split(" ");
677 				point2 = data[1].split(" ");
678 				line = new LineContent(new LocationContent(point1[0], point1[1]), 
679 									   new LocationContent(point2[2], point2[3]));
680 				list.push(line);
681 			}
682 		}
683 	}
684 	
685 	return list;
686 };

689  * Gets the polygons from the location of the POI object with a given term.
690  * @memberOf DataReader
691  * @param poi a single POI object (such as a Point Of Interest, Route or Event).
692  * @param term the wanted term.
693  * @returns {PolygonContent[]} an array containing all polygons with the given term or an empty array.
694  */
695 DataReader.getLocationPolygon = function(poi, term) {
696 	var list = [],
697 		location,
698 		polygon,
699 		polygons,
700 		data,
701 		posList,
702 		i;
703 	if(poi == null || poi == undefined)
704 		return list;
705 		
706 	location = poi.location;
707 	if(location.polygon != undefined) {
708 		polygons = location.polygon;
709 		for(key in polygons) {
710 			if(polygons[key].term == term) {
711 				polygon = new PolygonContent();
712 				data = [];
713 				data = polygons[key].SimplePolygon.posList.split(",");
714 				for(i = 0; i < data.length; i++) {
715 					posList = data[i].split(" ");
716 					polygon.addLocation(new LocationContent(posList[0], posList[1]));
717 				}
718 				list.push(polygon);
719 			}
720 		}
721 	}
722 	
723 	return list;

733 DataReader.getLocationGeometry = function(poi, term) {

736 		lineList,

739 		return list;

788  * Gets the categories in a given language.
789  * @memberOf DataReader
790  * @param poi the object to get the data.
791  * @param lang the wanted language.
792  * @returns {String} a list containing the following: the categories in the desired language if none found or empty.
793  */
794 DataReader.getCategories = function(poi, lang) {
795 	if(poi == null || poi == undefined)
796 		return false;
797 
798 	if(this.defaultLanguage == undefined)
799 		this.setDefaultLanguage("en_GB");
800 	
801 	var categoriesLang, poiLang = poi.lang,
802 		categories = poi.category,
803 		list = [];
804 	for(key in categories) {
805 		if(categories[key].lang == undefined) 
806 			categoriesLang = poiLang;
807 		else
808 			categoriesLang = categories[key].lang;
809 			
810 		if(this.langsAreEqual(categoriesLang, this.defaultLanguage)) {
811 			list.push(categories[key].value);
812 		}
813 	}
814 	
815 	return list;
816 };

839 		}
840 	}
