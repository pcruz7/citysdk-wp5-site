 26 var operator = {
 27 	"DEFAULT_OP": {
 28 		"value": "",
 29 		"operator": "",
 30 		"separator": ",",
 31 		"named": false
 32 	},
 33 	"PLUS_OP": {
 34 		"value": "+",
 35 		"operator": "",
 36 		"separator": ",",
 37 		"named": false
 38 	},
 39 	"SHARP_OP": {
 40 		"value": "#",
 41 		"operator": "#",
 42 		"separator": ",",
 43 		"named": false
 44 	},
 45 	"DOT_OP": {
 46 		"value": ".",
 47 		"operator": ".",
 48 		"separator": ".",
 49 		"named": false
 50 	},
 51 	"SLASH_OP": {
 52 		"value": "/",
 53 		"operator": "/",
 54 		"separator": "/",
 55 		"named": false
 56 	},
 57 	"SEMICOLON_OP": {
 58 		"value": ";",
 59 		"operator": ";",
 60 		"separator": ";",
 61 		"named": true
 62 	},
 63 	"QUERY_OP": {
 64 		"value": "?",
 65 		"operator": "?",
 66 		"separator": "&",
 67 		"named": true
 68 	},
 69 	"FORM_OP": {
 70 		"value": "&",
 71 		"operator": "&",
 72 		"separator": "&",
 73 		"named": true
 74 	}
 75 };

 77 /**
 78  * @name UriTemplate
 79  * @class Implementation of the URI Template RFC6570 (<a target="_blank" href="http://tools.ietf.org/html/rfc6570">http://tools.ietf.org/html/rfc6570</a>)
 80  */
 81 UriTemplate = function() {
 82 	var template, operators;
 83 };
 84 
 85 /**
 86  * Loads a given template
 87  * @memberOf UriTemplate
 88  * @param template to load
 89  */
 90 UriTemplate.fromTemplate = function(template) {
 91 	this.template = template;
 92 	this.operators = [];
 93 	this.operators[""] = operator.DEFAULT_OP;
 94 	this.operators["+"] = operator.PLUS_OP;
 95 	this.operators["#"] = operator.SHARP_OP;
 96 	this.operators["."] = operator.DOT_OP;
 97 	this.operators["/"] = operator.SLASH_OP;
 98 	this.operators[";"] = operator.SEMICOLON_OP;
 99 	this.operators["?"] = operator.QUERY_OP;
100 	this.operators["&"] = operator.FORM_OP;
101 };
102 
103 /**
104  * Checks whether a template has a given parameter
105  * @memberOf UriTemplate
106  * @param parameter check if a given template has a parameter
107  * @return {boolean} true if it is present, false otherwise
108  */
109 UriTemplate.hasParameter = function(parameter) {
110 	var templateRegex = /{[^{}]+}/g,
111 		varRegex = /((?:\w+))/g,
112 		match, vars;
113 	while (match = templateRegex.exec(this.template)) {
114 		template = match[0];
115 		while(vars = varRegex.exec(template)) {
116 			if(vars[0] == parameter)
117 				return true;
118 		}
119 	}
120 	
121 	return false;
122 };
123 
124 /**
125  * Converts a given loaded template into a URI containg the given values
126  * @memberOf UriTemplate
127  * @param values values of each parameter
128  * @return {String} the converted URI
129  */
130 UriTemplate.build = function(values) {
131 	var templateRegex = /{[^{}]+}/g,
132 		varRegex = /((?:\w+))/g,
133 		match, vars, 
134 		uri = this.template, 
135 		parameters = "",
136 		modifier,
137 		template,
138 		size, op;
139 	
140 	while (match = templateRegex.exec(this.template)) {
141 		template = match[0];
142 		op = this.operators[template.substring(1, 2)];
143 
144 		parameters += op.operator;
145 		while(vars = varRegex.exec(template)) {
146 			modifier = template.charAt(vars.index + vars[0].length);
147 			if(modifier == '*') {
148 				parameters += explode(vars[0], op);
149 			} else if(modifier == ':') {
150 				size = varRegex.exec(template);
151 				parameters += modify(vars[0], size[0], op);
152 			} else {
153 				parameters += expand(vars[0], op);
154 			}
155 		}
156 		
157 		if (op.separator && endsWith(parameters, op.separator))
158 				parameters = parameters.substring(0, parameters.length - 1);
159 
160 		uri = uri.replace(template, parameters);
161 	}
162 	
163 	return uri;
164 	
165 	function endsWith(str, suffix) {
166 	    return str.indexOf(suffix, str.length - suffix.length) !== -1;
167 	}
168 	
169 	function expand(vars, op) {
170 		var parameters = "";
171 		if(values[vars] != undefined) {
172 			if(isSimpleType(toType(values[vars]))) {
173 				parameters += expandSimple(vars, op, values[vars]);
174 			} else if(isArray(toType(values[vars]))) {
175 				parameters += expandArray(vars, op, values[vars]);
176 			}
177 		}
178 		
179 		return parameters;
180 	}
181 	
182 	function modify(vars, size, op) {
183 		var parameters = "", v;
184 		if(values[vars] != undefined) {
185 			v = values[vars].substring(0, parseInt(size));
186 			if (op.named)
187 				parameters += vars + "=" + v;
188 			else
189 				parameters += v;
190 
191 			parameters += op.separator;
192 		}
193 
194 		return parameters;
195 	}
196 	
197 	function explode(vars, op) {
198 		var parameters = "";
199 		if(values[vars] != undefined) {
200 			if(isSimpleType(toType(values[vars]))) {
201 				parameters += expandSimple(vars, op, values[vars]);
202 			} else if(isArray(toType(values[vars]))) {
203 				parameters += explodeArray(vars, op, values[vars]);
204 			}
205 		}
206 		
207 		return parameters;
208 	}
209 	
210 	function expandSimple(value, op, obj) {
211 		var parameters = "";
212 		if(op.named)
213 			parameters += value + "=" + obj;
214 		else
215 			parameters += obj;
216 		
217 		parameters += op.separator;
218 		return parameters;
219 	}
220 	
221 	function expandArray(value, op, array) {
222 		var parameters = "",
223 			separator = ",", ob;
224 		if(op.named)
225 			parameters += value + "=";
226 			
227 		for(key in array) {
228 			ob = array[key];
229 			if(!isNaN(key)) { // an indexed array
230 				parameters += ob.toString() + separator;
231 			} else { // an associative array
232 				parameters += key.toString() + "," + ob.toString() + separator;
233 			}
234 		}
235 		
236 		parameters = parameters.substring(0, parameters.length - 1);
237 		parameters += op.separator;
238 		return parameters;
239 	}
240 	
241 	function explodeArray(value, op, array) {
242 		var parameters = "",
243 			separator = op.separator, ob;
244 		
245 		for(key in array) {
246 			ob = array[key];
247 			if(!isNaN(key)) { // an indexed array
248 				if (op.named) {
249 					parameters += value + "=" + ob.toString() + separator;
250 				} else {
251 					parameters += ob.toString() + separator;
252 				}
253 			} else { // associative array
254 				if (op.named) {
255 					parameters += key.toString() + "=" + ob.toString() + separator;
256 				} else {
257 					parameters += key.toString() + op.separator + ob.toString() + separator;
258 				}
259 			}
260 		}
261 		
262 		parameters = parameters.substring(0, parameters.length - 1);
263 		parameters += op.separator;
264 		return parameters;
265 	}
266 	
267 	function isSimpleType(value) {
268 		return value == "string" 
269 			|| value == "number" 
270 			|| value == "boolean";
271 	}
272 	
273 	function isArray(value) {
274 		return value == "array";
275 	}
276 	
277 	function toType(obj) {
278  		return ({}).toString.call(obj).match(/\s([a-zA-Z]+)/)[1].toLowerCase();
279 	}
280 };
